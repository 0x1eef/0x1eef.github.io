def test_stdout_with_fork
  code = <<-CODE.each_line.map { _1.chomp.strip }.join(";")
    $stdout.sync = true
    pid = fork do
      sleep(1)
      puts "bar"
    end
    puts "foo"
    Process.wait(pid)
  CODE
  assert_equal "foo\nbar\n", cmd(%Q(ruby -e '#{code}')).stdout
end
