Overview

The Server class provides a rack application that serves the contents of a directory over HTTP. The underlying server is puma.

def initialize(app, options = {})
  @app = app
  @options = default_options.merge!(options)
  @events = Puma::Events.new
  @server = Puma::Server.new(@app, @events, @options)
end

def self.prepare(options)
  options = options.to_h.dup
  host = options.delete(:host)  ||
         options.delete("host") ||
         options.delete(:bind)  ||
         options.delete("bind") ||
         "127.0.0.1"
  port = options.delete(:port)  ||
         options.delete("port") ||
         3000
  unix = options.delete(:unix)  ||
         options.delete("unix") ||
         nil
  options.merge!(
    binds: [unix ? "unix://#{unix}" : "tcp://#{host}:#{port}"]
  )
  options
end

def start(block: false)
  @server.binder.parse(@options[:binds])
  thr = @server.run
  block ? thr.join : thr
end
