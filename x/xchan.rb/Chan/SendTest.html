def test_send_race_condition
  pids = 4.times.map { fork { exit(ch.recv.to_i) } }
  sleep(0.1 * 4)
  pids.each.with_index(1) { ch.send(object) }
  assert_equal pids.map { 42 }, pids.map { Process.wait2(_1).last.exitstatus }
end
