def call(*args)
  return if @pid != Process.pid

  $stderr.puts "removing #{@tmpfile.path}..." if $DEBUG

  @tmpfile.close
  begin
    File.unlink(@tmpfile.path)
  rescue Errno::ENOENT
  end

  $stderr.puts "done" if $DEBUG
end
