def call(...)
  return super unless @tracer
  span = @tracer.on_tool_start(id:, name:, arguments:, model:)
  result = super
  @tracer.on_tool_finish(result:, span:)
  result
rescue => ex
  @tracer.on_tool_error(ex:, span:)
  raise(ex)
end
