Overview

Coordinates shared access to a transport by routing JSON-RPC responses to the mailbox waiting on the matching request id.

def register
  @lock.synchronize do
    @request_id += 1
    mailbox = LLM::MCP::Mailbox.new
    @pending[@request_id] = mailbox
    [@request_id, mailbox]
  end
end

def route(response)
  mailbox = @lock.synchronize { @pending[response["id"]] }
  raise LLM::MCP::MismatchError.new(expected_id: nil, actual_id: response["id"]) unless mailbox
  mailbox << response
  nil
end
