def parse_tool_input(input)
  case input
  when Hash then input
  when String
    parsed = LLM.json.load(input)
    Hash === parsed ? parsed : {}
  when nil then {}
  else
    input.respond_to?(:to_h) ? input.to_h : {}
  end
rescue *LLM.json.parser_error
  {}
end
