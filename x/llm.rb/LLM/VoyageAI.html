def embed(input, model: "voyage-2", **params)
  req = Net::HTTP::Post.new("/v1/embeddings", headers)
  req.body = JSON.dump({input:, model:}.merge!(params))
  res = request(@http, req)
  Response::Embedding.new(res).extend(response_parser)
end
