def run(content, options = {})
  options = Ryo.from(options)
  depend_on dependable(paths: options.depend_on, reject: options.reject)
            .map { items[_1] }
  webpack temporary_file(content),
          cli: self.class.default_options.merge(options.cli || {})
end
