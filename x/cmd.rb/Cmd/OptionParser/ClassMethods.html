def option_parser
  @option_parser ||= ::OptionParser.new(nil, 26, " " * 2).tap do
    _1.banner = ""
    _1.on("-h", "--help", "Show help")
  end
end

def set_option(short, long, desc, as: String, default: nil)
  option_parser.on(short, long, desc, as)
  set_default({ short => default, long => default })
