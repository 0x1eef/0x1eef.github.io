def test_ls_command_help
  help = "Usage: ls [OPTIONS]\n\n" \
         "Description:\n" \
         "  Lists the contents of a directory\n\n" \
         "Options: \n" \
         "  -h, --help                 Show help\n" \
         "  -g, --grep PATTERN         A regular expression\n" \
         "  -p, --path PATH            The target path\n\n"
  assert_equal help, run_example("ls --help").stdout
end
